–ù–∞: –¥–æ–ª—è '—É—Å–ø–µ—Ö–∞' –≤ –ø–æ–ø—É–ª—è—Ü–∏–∏ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–∞–ª—å–Ω–æ–≥–æ –ª–µ—á–µ–Ω–∏—è **–±–æ–ª—å—à–µ** –¥–æ–ª–∏ '—É—Å–ø–µ—Ö–∞' –≤ –ø–æ–ø—É–ª—è—Ü–∏–∏ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–≥–æ –ª–µ—á–µ–Ω–∏—è.

$–ù_–∞:$ $p_{—ç–∫—Å–ø. –ª–µ—á} > p_{—Å—Ç. –ª–µ—á}$

##### –£–∫–∞–∂–∏—Ç–µ —É—Ä–æ–≤–µ–Ω—å –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏ Œ±.

–ó–∞–¥–∞–¥–∏–º —É—Ä–æ–≤–µ–Ω—å –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏ $\alpha = 0.05$.

##### –û–ø—Ä–µ–¥–µ–ª–∏—Ç–µ —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏–π —Ç–µ—Å—Ç (–∫—Ä–∏—Ç–µ—Ä–∏–π) –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –≥–∏–ø–æ—Ç–µ–∑—ã. –ü—Ä–∏ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –∞—Ä–≥—É–º–µ–Ω—Ç–∏—Ä—É–π—Ç–µ –≤—ã–±–æ—Ä –æ–¥–Ω–æ—Å—Ç–æ—Ä–æ–Ω–Ω–µ–≥–æ —Ç–µ—Å—Ç–∞.

–í—ã–±–∏—Ä–∞–µ–º Z-–∫—Ä–∏—Ç–µ—Ä–∏–π –¥–ª—è —Ä–∞–∑–Ω–æ—Å—Ç–∏ –¥–æ–ª–µ–π –¥–ª—è –Ω–µ–∑–∞–≤–∏—Å–∏–º—ã—Ö –≤—ã–±–æ—Ä–æ–∫. –¢a–∫ –∫–∞–∫ –Ω–∞–±–ª—é–¥–µ–Ω–∏–π –≤ –∫–∞–∂–¥–æ–π –≥—Ä—É–ø–ø–µ –±–æ–ª—å—à–µ 5, —Ç–æ –≤—ã–±–∏—Ä–∞–µ–º –∞—Å–∏–º–ø—Ç–æ—Ç–∏—á–µ—Å–∫–∏–π —Ç–µ—Å—Ç. –¢–ê–∫ –∫–∞–∫ –≥–æ–≤–æ—Ä–∏—Ç—Å—è –æ **—É–≤–µ–ª–∏—á–µ–Ω–∏–∏ –¥–æ–ª–∏**, —Ç–æ –∏—Å–ø–æ–ª—å–∑—É–µ–º –æ–¥–Ω–æ—Å—Ç–æ—Ä–æ–Ω–Ω–∏–π —Ç–µ—Å—Ç(–ø—Ä–∞–≤–æ—Å—Ç–æ—Ä–æ–Ω–Ω–∏–π)

##### –£–∫–∞–∂–∏—Ç–µ –∫—Ä–∏—Ç–∏—á–µ—Å–∫–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∏ (–∏—Å–ø–æ–ª—å–∑—É–π—Ç–µ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –∏–∑ –º–∞—Ç–µ—Ä–∏–∞–ª–æ–≤ –ª–µ–∫—Ü–∏–∏).

``` r
# —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ z-–∫—Ä–∏—Ç–µ—Ä–∏—è —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∞ –Ω–æ—Ä–º–∞–ª—å–Ω–æ, —Å–æ —Å—Ä–µ–¥–Ω–∏–º –∑–Ω–∞—á–µ–Ω–∏–µ–º 0 –∏ –¥–∏—Å–ø–µ—Ä—Å–∏–µ–π 1.
```

``` r
# –ù–∞–ø–∏—à–∏–º —Ñ—É–Ω–∫—Ü–∏—é –¥–ª—è –ø—Ä–æ–≤–µ–¥–µ–Ω–∏—è —Ç–µ—Å—Ç–∞
# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–æ–≤–µ–¥–µ–Ω–∏—è z-—Ç–µ—Å—Ç–∞ –¥–ª—è –¥–≤—É—Ö –Ω–µ–∑–∞–≤–∏—Å–∏–º—ã—Ö –¥–æ–ª–µ–π
performTwoProportionZTest <- function(n1, x, n2, y) {
  # –†–∞—Å—á–µ—Ç –¥–æ–ª–µ–π —É—Å–ø–µ—à–Ω—ã—Ö –∏—Å—Ö–æ–¥–æ–≤ –≤ –≤—ã–±–æ—Ä–∫–∞—Ö
  p1 <- x / n1
  p2 <- y / n2

  # –†–∞—Å—á–µ—Ç –æ–±—ä–µ–¥–∏–Ω–µ–Ω–Ω–æ–π –¥–æ–ª–∏ —É—Å–ø–µ—à–Ω—ã—Ö –∏—Å—Ö–æ–¥–æ–≤
  P <- (x + y) / (n1 + n2)

  # –†–∞—Å—á–µ—Ç —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–π –æ—à–∏–±–∫–∏
  SQ <- P * (1 - P) * (1/n1 + 1/n2)

  # –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ z-—Ç–µ—Å—Ç–∞
  z_test_result <- (p1 - p2) / sqrt(SQ)

  return(z_test_result)
}

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏:
n1 <- 100  # –û–±—â–µ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–æ–ø—ã—Ç–æ–∫ –¥–ª—è —ç–∫—Å–ø. –ª–µ—á

z_test_result <- performTwoProportionZTest(n1, x, n2, y)
print(z_test_result)
```

``` r
# –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏–µ p_value –¥–ª—è –æ–¥–Ω–æ—Å—Ç–æ—Ä–æ–Ω–Ω–µ–≥–æ —Ç–µ—Å—Ç–∞
p_value <- 1 -  pnorm(z_test_result)




# –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏–µ 95% –¥–ª—è –æ–¥–Ω–æ—Å—Ç–æ—Ä–æ–Ω–Ω–µ–≥–æ —Ç–µ—Å—Ç–∞ —Ñ–æ—Ä–º—É–ª–∞ CI=[(x‚àíy)‚àíZŒ±/2*SQ,+‚àû]

SQ <- function(n1, x, n2, y) {
  p1 <- x / n1
  p2 <- y / n2

  # –†–∞—Å—á–µ—Ç –æ–±—ä–µ–¥–∏–Ω–µ–Ω–Ω–æ–π –¥–æ–ª–∏ —É—Å–ø–µ—à–Ω—ã—Ö –∏—Å—Ö–æ–¥–æ–≤
  P <- (x + y) / (n1 + n2)

  # –†–∞—Å—á–µ—Ç —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–π –æ—à–∏–±–∫–∏
  SQ <- P * (1 - P) * (1/n1 + 1/n2)
return(SQ)
  }
```

## –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –¥–∞–Ω–Ω—ã—Ö


``` r
# –í–æ—Å–ø—Ä–æ–∏–∑–≤–æ–¥–∏–º–æ—Å—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
```

$–ù_0$: $p(M_{group1} > M_{group2}) = p(M_{group1} < M_{group2})$

$–ù_0$: $Med_{group1} = Med_{group2}$

$–ù_a$: $Med_{group1} \neq Med_{group2}$

–ó–∞–¥–∞–¥–∏–º —É—Ä–æ–≤–µ–Ω—å –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏ $\alpha = 0.05$.
